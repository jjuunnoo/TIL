# 예외 처리

```python
# try:
#     예외가 발생 할 수 있는 실행문
# except:
#     예외가 발생한 경우 실행되는 실행문
# else:
#     예외가 발생하지 않았을 때 실행되는 실행문
# finally:
#     무조건 실행되는 실행문

# 에러나는 for문
>>> nums = ["10.22", "" ,"8.00"]
>>>
for num in nums:
    print(float(num))

ValueError: could not convert string to float: ''    
        
# try, except 사용
>>> nums = ["10.22", "" ,"8.00"]
>>>
for num in nums:
    try:
        print(float(num))
    except Exception as e:
        print(0)
    else:
        print('변환 성공')
    finally:
        print('변환 완료')
        
10.22
변환 성공
변환 완료
could not convert string to float: ''
변환 완료
8.0
변환 성공
변환 완료
```



# 함수 몇가지

```python
# abs: 절댓값
>>> abs(-4)
4

# all, 전부 True 일때 True 출력
>>> all([1, 2, 3, 4])
True
>>> all([1, 2, 3, 0])
False

# any: 하나라도 True 이면 True
>>> any([1, 2, 3, 0])
True
>>> any([0, ""])
False

# chr: 아스키 코드에 맞는 문자 출력
>>> chr(97)
'a'

# ord: 문자 값에 맞는 아스키코드 출력
>>> ord('a')
97

# divmod 몫과 나머지 튜플로 출력
>>> divmod(7,3)
(2, 1)

# enumerate: 인덱스 필요할 때 사용
>>>
for i , name in enumerate(['가', '나', '다']):
    print(i, name)   
    
0 가
1 나
2 다    

# eval: 문자열 자체를 수식으로 실행
>>> eval('1+2')    
3
>>> eval('divmod(7,3)')
(2, 1)

# filter(함수, iterable)
# iterable 객체를 하나하나 빼서 함수에 적용
# 리턴값을 받아서 filter 결과값이 True 인것만 filter 객체로 출력됨 
# ex)
# 
>>>
def positive(x):
    return x>0
>>> list(filter(positive, [1, -2, -3, 4, 5]))
[1, 4, 5]

# isinstance: 객체가 해당 클래스인지 확인
>>>
class Person:
    pass
>>> a = Person()
>>> isinstance(a, Person)
True
>>> isinstance(a, str)
False

# map: 키역할을 하는 데이터와 값역할을 하는 데이터를 하나씩 짝지어 저장하는 구조
>>>
def x22(x):
    return x*2
>>> list(map(x22, [1,2,3,4]))
[2, 4, 6, 8]

# pow 거듭제곱
>>> pow(2, 4)
16

# round 반올림
>>> round(5.5)
6

# sorted 정렬
>>> sorted([3, 1, 5, 2])
[1, 2, 3, 5]

# zip
# 같은 인덱스를 가지는 값끼리 묶어줌 tuple로 묶어줌
>>> list(zip([1, 2, 3], [4, 5, 6]))
[(1, 4), (2, 5), (3, 6)]
>>> list(zip([1,2,3,4],[5,6]))
[(1, 5), (2, 6)]
>>> list(zip('abcdefg', [5,6,7], range(5)))
[('a', 5, 0), ('b', 6, 1), ('c', 7, 2)]
```
