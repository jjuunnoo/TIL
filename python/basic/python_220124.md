>>> import pandas as pd

##################### series #######################
# 1차원 데이터
# 인덱스, 밸류로 이루어져있음 
# array-like, iterable, dict, scalar value 들어갈 수 있음

>>> s = pd.Series([3, 5, 7, 9])
>>> s
0    3
1    5
2    7
3    9
dtype: int64
    
>>> s2 = pd.Series([3, 5, 7, 9], index = ['a', 'b', 'c', 'd'])
>>> s2    
a    3
b    5
c    7
d    9
dtype: int64
    
# 시리즈의 인덱스 확인
>>> s.index
RangeIndex(start=0, stop=4, step=1)
>>> s2.index
Index(['a', 'b', 'c', 'd'], dtype='object')

# 시리즈의 값 확인
>>> s.values
array([3, 5, 7, 9], dtype=int64)
