# 제어문

- if문

  ```python
  # 예시
  
  # bool
  >>> isNormal = True
  >>> if isNormal:
      	print('정상')
      else:
          print('비정상')
  정상
  
  # or
  # 확률적으로 참일 가능성 높은 조건을 앞에 배치하는 것이 좋다
  # 참을 확인하면 뒤의 조건을 확인 할 필요가 없기 때문
  >>> x = 30
  >>> is_y = True
  >>> if x > 50 or is_y:
          print('조건 만족')
      else:
          print('불만족')  
  조건 만족
  
  # and
  # 확률적으로 거짓일 가능성 높은 조건을 앞에 배치하는 것이 좋다
  # 거짓을 확인하면 뒤의 조건을 확인 할 필요가 없기 때문
  >>> x = 30
  >>> is_y = True
  >>> if x > 50 and is_y:
          print('조건 만족')
      else:
          print('불만족')    
  불만족
  
  # elif
  >>> x = 30
  >>> is_y = True
  >>> if x > 50:
          print('조건 만족')
      elif is_y:
          print('y 만족')    
      else:
          print('불만족') 
  y 만족
  
  # 한줄 조건문
  >>> score = 80
  >>> result = 'good' if score >=80 else 'fail'
  >>> print(result)
  good
  ```

  

- while문

  ```python
  # 예시
  >>> tryCnt = 0
  >>> while tryCnt < 10:

          print(f'{tryCnt}', end=' ')
          if tryCnt >=10:

  1. 추가
  2. 삭제
  3. 보기

  """
  >>> num = 0

  2. 삭제
  3. 보기
  4. 그만두기
  번호 입력: # 4 입력시 종료

  >>> num_coffee = 3
  	price_coffee = 5000
      prompt_1 = f'''
      커피 판매중 {price_coffee}원

      2. 종료

      prompt_2 = '''
      돈 투입:'''
      while num_coffee:
          print(f'{prompt_1}')

              print(f'{prompt_2}')
  ```

- for문

  ```python
  # for in 리스트/튜플/문자열 가능
  >>> LCnt = ['one', 'two', 'three']
  >>> for i in LCnt:
      	print(i, end=' ')
  one two three
  
  >>> aaa = 'apple'
      for i in aaa:
          print(i, end='')
  apple
  
  # range()
  # range([start], stop, [step]) / start = 0, step = 1 기본값
  
  >>> result = 0
  >>> for i in range(1,11):
          result += i
      print(result)
  55
  
  # List Comprehension
  >>> scores = [98, 70, 85, 65, 50]
  >>> result = [num + 3 for num in scores]
  >>> result
  [101, 73, 88, 68, 53]
  
  # List Comprehension 안에 조건문
  >>> scores = [98, 70, 85, 65, 50]
  >>> result = [num + 3 for num in scores if num < 97]
  >>> result
  [73, 88, 68, 53]
  
  # 같은 for문 다른 표현법
  >>> result_1 = [x * y for x in range(2,4) for y in range(1,4)]
  >>> result_2 = []
      for x in range(2,4):
          for y in range(1,4):
              result_2.append(x*y)
  >>> result_1, result_2
  ([2, 4, 6, 3, 6, 9], [2, 4, 6, 3, 6, 9])
  ```

  

  # 함수

  ```python
  # 일반 형태
  >>> def add_func(a, b):
      	return a+b
  >>> add_func(3,5)
  8
  
  # 입력 값이 없는 경우
  # 아래 예시의 경우 입력값은 없고 출력값만 존재
  >>> def say():
      	return 'hi'
  >>> a = say()
  >>> a
  'hi'
  
  # 결과 값이 없는 경우
  # 아래 예시의 경우 화면에 출력 할 뿐, 출력값이 없다
  >>> def add_func2(a, b):
  	    print(f'{a}, {b}의 합은 {a+b}')
  >>> add_func2(5,6)
  5, 6의 합은 11
  
  # 입력값, 출력값 모두 없는 경우
  # 아래 예시의 경우 입력값, 출력값이 모두 없으며 화면에 출력만한다.
  >>> def say():
      	print('hi')
  >>> say()
  hi
  ```
